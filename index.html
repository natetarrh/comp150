Given a grammar,

Grammar types


type Grammar n t = (Set n, n, t -> Set n, Set (Production n))

type Production n = (n, [n])

data ParseTree n t = Leaf t | n :^ [ParseTree n t]

parse :: Grammar n t -> [t] -> [ParseTree n t]

CYK algorithm

Yikes!

    grammar :: [Rule]
    parse :: STUArray s (Int, Int, Int) Rule

    forM_ (enumerate input) $ \(i, t) ->
      forM_ (terminalRules grammar) $ \(j, rule) ->
        when (terminalRuleProduces rule t) $
          writeArray parse (i,1,j) rule
 
    forM_ [2..n] $ \i ->
      forM_ [1..(n-i+1)] $ \j ->
        forM_ [1..(i-1)] $ \k ->
          forM_ (nonTerminalRules grammar) $ \(a, rule) ->
            forM_ (lookup indexMap rule) $ \(b,c) -> do
              p0 <- readArray parse (j,k,b)
              p1 <- readArray parse (j+k,i-k,c)
                when (p0 && p1) $
                  writeArray parse (j,i,a) rule
